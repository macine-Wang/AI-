### 主要接口

// 候选人
interface Candidate {
  id: string;
  name: string;
  email: string;
  phone: string;
  position: string;
  source: string;
  status: 'new' | 'screening' | 'interview' | 'offer' | 'hired' | 'rejected';
  matchScore: number;
  skills: string[];
  experience: string;
  education: string;
  expectedSalary?: number;
  currentCompany?: string;
  resumeUrl?: string;
  appliedAt: Date;
  tags: string[];
  notes: string;
  interviews: Interview[];
  communications: Communication[];
}

// 面试
interface Interview {
  id: string;
  date: Date;
  interviewer: string;
  round: number;
  type: string;
  score?: number;
  feedback?: string;
  status: 'scheduled' | 'completed' | 'cancelled';
}

// 沟通记录
interface Communication {
  id: string;
  type: 'email' | 'phone' | 'message';
  content: string;
  sendAt: Date;
  template?: string;
}

// 邮件模板
interface EmailTemplate {
  id: string;
  name: string;
  subject: string;
  content: string;
  type: 'invite' | 'reject' | 'offer' | 'reminder' | 'followup';
}

// 统计数据
interface RecruitmentStats {
  totalCandidates: number;
  newApplications: number;
  interviewScheduled: number;
  offersExtended: number;
  hired: number;
  avgMatchScore: number;
  avgProcessTime: number;
  conversionRate: number;
}

// AI消息
interface MessageItem {
  id: string;
  type: 'user' | 'ai';
  content: string;
  timestamp: Date;
}
